% Social Sciences / Humanities (APA) caption style per UC3M Plantilla_TFM_ingles_2019
% (verified against the rendered ModeloTFM_CCSS-HUM exemplar): table captions ABOVE,
% bold, left-aligned ("Table N. Title"); figure captions BELOW, left-aligned,
% regular weight ("Figure N. Title").
\captionsetup[table]{
  justification=raggedright,
  singlelinecheck=false,
  labelsep=period,
  labelfont=small,
  font={small,bf}
}

\captionsetup[figure]{
  format=hang,
  name=Figure,
  justification=raggedright,
  singlelinecheck=false,
  labelsep=period,
  labelfont=small,
  font=small
}
