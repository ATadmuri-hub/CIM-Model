Scrubber = function (values, {format = v=>v, initial=0, direction=1, delay=null, autoplay=true, loop=true, alternate=false} = {}) {
  values = Array.from(values);
  const form = html`<form style="font:12px sans-serif;display:flex;height:33px;align-items:center;">
    <button name=b type=button style="margin-right:.4em;width:5em;">${autoplay?"Pause":"Play"}</button>
    <label style="display:flex;align-items:center;">
      <input name=i type=range min=0 max=${values.length-1} value=${initial} step=1 style="width:220px;">
      <output name=o style="margin-left:.5em;"></output>
    </label></form>`;
  let frame=null, interval=null;
  function start(){ form.b.textContent="Pause"; if(delay===null) frame=requestAnimationFrame(tick); else interval=setInterval(tick,delay); }
  function stop(){ form.b.textContent="Play"; if(frame!==null)cancelAnimationFrame(frame),frame=null; if(interval!==null)clearInterval(interval),interval=null; }
  function running(){ return frame!==null||interval!==null; }
  function tick(){ if(form.i.valueAsNumber===(direction>0?values.length-1:0)){ if(!loop) return stop(); if(alternate) direction=-direction; } if(delay===null) frame=requestAnimationFrame(tick); step(); }
  function step(){ form.i.valueAsNumber=(form.i.valueAsNumber+direction+values.length)%values.length; form.i.dispatchEvent(new CustomEvent("input",{bubbles:true})); }
  form.i.oninput=event=>{ if(event&&event.isTrusted&&running())stop(); form.value=values[form.i.valueAsNumber]; form.o.value=format(form.value, form.i.valueAsNumber, values); };
  form.b.onclick=()=>{ if(running())return stop(); direction=alternate&&form.i.valueAsNumber===values.length-1?-1:direction; form.i.valueAsNumber=(form.i.valueAsNumber+direction)%values.length; form.i.dispatchEvent(new CustomEvent("input",{bubbles:true})); start(); };
  form.i.oninput();
  if(autoplay) start(); else stop();
  Inputs.disposal(form).then(stop);
  return form;
}

{
  const a = traj[scen];
  const base = traj["Baseline"];
  function tchart(metric) {
    const m = metricMeta[metric];
    const mk = metric+"_mean", sk = metric+"_sd";
    const rows = a.week.map((w,i)=>({week:w, y:a[mk][i], lo:a[mk][i]-(a[sk][i]||0), hi:a[mk][i]+(a[sk][i]||0)})).filter(d=>d.week<=week);
    const bl = base.week.map((w,i)=>({week:w, y:base[mk][i]})).filter(d=>d.week<=week);
    const marks = [
      Plot.areaY(rows,{x:"week",y1:"lo",y2:"hi",fill:m.color,fillOpacity:0.15}),
      Plot.ruleX([9],{stroke:"#ccc",strokeDasharray:"3,3"})
    ];
    if (scen !== "Baseline") marks.push(Plot.lineY(bl,{x:"week",y:"y",stroke:"#9a9a9a",strokeWidth:1.2,strokeDasharray:"3,2"}));
    marks.push(Plot.lineY(rows,{x:"week",y:"y",stroke:m.color,strokeWidth:2}));
    marks.push(Plot.text([{w:rows.length?rows[rows.length-1].week:0, y:rows.length?rows[rows.length-1].y:0}],{x:"w",y:"y",text:d=>d3.format(".2f")(d.y),dx:6,fontSize:10,fill:m.color}));
    marks.push(Plot.text([m.label],{frameAnchor:"top-left",dy:-14,fontSize:11,fontWeight:"bold"}));
    return Plot.plot({height:215, width:300, marginLeft:40, marginBottom:28, x:{domain:[0,52],label:"week"}, y:{label:null,nice:true}, marks});
  }
  const grid=html`<div style="display:grid;grid-template-columns:repeat(3,1fr);gap:10px"></div>`;
  for(const k of Object.keys(metricMeta)) grid.append(tchart(k));
  return grid;
}

viewof weekS = Scrubber(d3.range(0,53), {autoplay:false, delay:160, loop:true, format:w=>"Week "+w})
World view: ▲ park · ★ trainer · ● migrant (red) / local (blue), brightness = motivation · grey = dropped · orange = winter-paused · orange link = cross-group tie 
{
  const r=replay, W=600, H=600, pad=22, sc=(W-2*pad)/r.world;
  const SX=v=>pad+v*sc, SY=v=>pad+(r.world-v)*sc;
  const pos=new Map(r.id.map((id,i)=>[id,{x:SX(r.x[i]),y:SY(r.y[i]),kind:r.kind[i]}]));
  const svg=d3.create("svg").attr("viewBox",[0,0,W,H]).attr("width",W).attr("height",H).attr("style","max-width:100%;height:auto;background:#fff;border:1px solid #e3e3e3;font:11px sans-serif");
  const E=r.edges[weekS].filter(e=>pos.has(e[0])&&pos.has(e[1]));
  svg.append("g").selectAll("line").data(E).join("line").attr("x1",e=>pos.get(e[0]).x).attr("y1",e=>pos.get(e[0]).y).attr("x2",e=>pos.get(e[1]).x).attr("y2",e=>pos.get(e[1]).y).attr("stroke",e=>+e[2]?pal.orange:pal.grey).attr("stroke-width",e=>+e[2]?2:0.7).attr("stroke-opacity",0.8);
  const mot=r.mot[weekS], stt=r.state[weekS];
  const g=svg.append("g").selectAll("g").data(d3.range(r.id.length)).join("g").attr("transform",i=>`translate(${pos.get(r.id[i]).x},${pos.get(r.id[i]).y})`);
  g.each(function(i){ const k=r.kind[i], s=stt[i], m=mot[i], sel=d3.select(this);
    if(k==="park"){ sel.append("text").text("▲").attr("text-anchor","middle").attr("dy",5).attr("font-size",18).attr("fill","#2e7d32"); return; }
    if(k==="trainer"){ sel.append("text").text("★").attr("text-anchor","middle").attr("dy",5).attr("font-size",15).attr("fill",pal.gold); return; }
    let col=s===2?"#bdbdbd":s===1?pal.orange:(k==="migrant"?d3.interpolateRgb("#7a0010","#ff8a8a")(m):d3.interpolateRgb("#0a2a44","#7fc0ec")(m));
    sel.append("circle").attr("r",s===2?3:4.5).attr("fill",col).attr("stroke","#fff").attr("stroke-width",0.6);
  });
  const active=stt.filter((s,i)=>s===0&&(r.kind[i]==="migrant"||r.kind[i]==="local")).length;
  const dropped=stt.filter(s=>s===2).length, cross=E.filter(e=>+e[2]).length;
  svg.append("text").attr("x",12).attr("y",18).attr("font-weight","bold").text(`Week ${weekS}   ·   active ${active}   ·   dropped ${dropped}   ·   cross-group ties ${cross}`);
  return svg.node();
}
